(целого типа) определяется следующим образом:
F1 = 1,
F2 = 1,
FK = FK–2 + FK–1,
K = 3, 4, ... .
Вывести элементы F1, F2, ..., FN.
***

+ for N in range(5,20,4):
   +  print("N = ",N)
   +  F1 = 1
   +  F2 = 1
   +  print(1," : ",F1)
   +  print(2," : ",F2)
   +  for k in range(3,N+1):
        F3 = F2 + F1
        print(k," : ",F3)
        F1 = F2
